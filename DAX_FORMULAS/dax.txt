Average_Yield =

    'FAOSTAT_data_en_7-31-2025'[Element] = "Yield"

--------------------------------------------------------------------------------

Growth_Rate = 
VAR thisYear = 
    CALCULATE(
        SUM('FAOSTAT_data_en_7-31-2025'[Value]),
        'FAOSTAT_data_en_7-31-2025'[Element] = "Production",
        'FAOSTAT_data_en_7-31-2025'[Year] = MAX('FAOSTAT_data_en_7-31-2025'[Year])
    )
VAR lastYear = 
    CALCULATE(
        SUM('FAOSTAT_data_en_7-31-2025'[Value]),
        'FAOSTAT_data_en_7-31-2025'[Element] = "Production",
        'FAOSTAT_data_en_7-31-2025'[Year] = MAX('FAOSTAT_data_en_7-31-2025'[Year]) - 1
    )
RETURN
    DIVIDE(thisYear - lastYear, lastYear, 0)
